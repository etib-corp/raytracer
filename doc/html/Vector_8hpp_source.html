438inline Math::Vector<3> refract(const Math::Vector<3> &uv, const Math::Vector<3> &n, double etai_over_etat)
439{
440 double cos_theta = fmin((uv * -1.0).dot(n), 1.0);
441 Math::Vector<3> r_out_perp = (uv + n * cos_theta) * etai_over_etat;
442 Math::Vector<3> r_out_parallel = n * -sqrt(fabs(1.0 - r_out_perp.length_squared()));
443 return r_out_perp + r_out_parallel;
444}

338 [[nodiscard]] PMaterial withCylindreMaterialGet(const ui32 index)
339 {
340 std::string sType;
341 PMaterial pcylindre;
342 const Setting& root = getRoot();
343 const Setting& primitives = root["primitives"];
344 const Setting& cylindres = primitives["cylindres"];
345 const Setting& cylindre = cylindres[index];
346 const Setting& material = cylindre.lookup("material");
347
348 material.lookupValue("type", sType);
349 pcylindre.nameSet(sType);
350
351 if (material.exists("textures"))
352 {
353 PTexture ptexture(sType);
354 d64 scale = 0.0;
355 const Setting& textures = material.lookup("textures");
356
357 pcylindre.istextureSet(true);
358 textures.lookupValue("scale", scale);
359 ptexture.scaleSet(scale);
360
361 const Setting& colors = textures.lookup("colors");
362 for (ui32 i = 0; i != colors.getLength(); ++i) {
363 d64 rgb[3] = {0};
364 colors[i].lookupValue("r", rgb[0]);
365 colors[i].lookupValue("g", rgb[1]);
366 colors[i].lookupValue("b", rgb[2]);
367 ptexture.colorsSet((Math::Vector<3>({rgb[0], rgb[1], rgb[2]})));
368 }
369 pcylindre.addTexture(ptexture);
370 }
371 if (material.exists("color"))
372 {
373 d64 rgb[3] = {0.0};
374 const Setting& color = material.lookup("color");
375 pcylindre.isColorSet(true);
376 color.lookupValue("r", rgb[0]);
377 color.lookupValue("g", rgb[1]);
378 color.lookupValue("b", rgb[2]);
379 pcylindre.addColor(PColor(sType, Math::Vector<3>({rgb[0], rgb[1], rgb[2]})));
380 }
381 if (material.exists("refractive"))
382 {
383 pcylindre.isRefractiveSet(true);
384 d64 refractive_index = 0.0;
385 const Setting& color = material.lookup("refractive");
386 color.lookupValue("refractive_index", refractive_index);
387 pcylindre.addRefractive(PRefractive(sType, refractive_index));
388 }
389 return std::move(pcylindre);
390 }
